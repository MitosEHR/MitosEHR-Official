01034     {
01035         $curdir = str_replace('\\', '/', dirname($file));
01036         $path   = str_replace('\\', '/', $path);
01037         if (strpos($path, ':') !== false) {
01038             // windows, and we have a drive letter
01039             return $path;
01040         } elseif (strpos($path, '/') === 0) {
01041             return $path;
01042         }
01043         // not an absolute path
01044         $path = explode('/', $path);
01045         if ($path[0] == '.') {
01046             $path[0] = dirname($file);
01047             return join($path, '/');
01048         } elseif ($path[0] == '..') {
01049             $dirfile = explode('/', dirname(str_replace('\\', '/', $file)));
01050             // remove the current directory
01051             array_pop($dirfile); 
01052             if (!count($dirfile)) {
01053                 // we were at a top-level dir!
01054                 return false;
01055             }
01056             // replace .. with parent dirname
01057             $path[0] = join($dirfile, '/');
01058             return join($path, '/');
01059         } else {
01060             $path = join($path, '/');
01061             return array($curdir . PATH_DELIMITER . $path,
01062                 str_replace('\\', '/', PHPDOCUMENTOR_BASE)
