01751     function _read_file($filename, $start=null, $lines=null)
01752     {
01753         if (!($fd = @fopen($filename, 'r'))) {
01754             return false;
01755         }
01756         flock($fd, LOCK_SH);
01757         if ($start == null && $lines == null) {
01758             // read the entire file
01759             $contents = fread($fd, filesize($filename));
01760         } else {
01761             if ( $start > 1 ) {
01762                 // skip the first lines before $start
01763                 for ($loop=1; $loop < $start; $loop++) {
01764                     fgets($fd, 65536);
01765                 }
01766             }
01767             if ( $lines == null ) {
01768                 // read the rest of the file
01769                 while (!feof($fd)) {
01770                     $contents .= fgets($fd, 65536);
01771                 }
01772             } else {
01773                 // read up to $lines lines
01774                 for ($loop=0; $loop < $lines; $loop++) {
01775                     $contents .= fgets($fd, 65536);
01776                     if (feof($fd)) {
01777                         break;
01778                     }
01779                 }
01780             }
01781         }
01782         fclose($fd);
01783         return $contents;
01784     }
