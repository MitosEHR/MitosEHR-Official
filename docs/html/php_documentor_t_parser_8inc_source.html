02356     function retrieveType($value, $checkforvar = false)
02357     {
02358         if (!count($value)) {
02359             return false;
02360         }
02361         $result = array();
02362         $types  = '';
02363         // remove empty entries resulting from extra spaces between @tag and type
02364         $this->_removeWhiteSpace($value, 0);
02365         $index = 0;
02366         if (trim($value[0]) == 'object') {
02367             $types .= array_shift($value) . ' ';
02368             $this->_removeWhiteSpace($value, 0);
02369             if (!count($value)) {
02370                 // was just passed "object"
02371                 $result = array('type' => rtrim($types), 'desc' => '');
02372                 if ($checkforvar) {
02373                     $result['var'] = false;
02374                 }
02375                 return $result;
02376             }
02377             if ($value[0]{0} == '$' || substr($value[0], 0, 2) == '&$') {
02378                 // was just passed "object" and the next thing is a variable name
02379                 $result['var']  = trim($value[0]);
02380                 $result['type'] = 'object';
02381                 array_shift($value);
02382                 $result['desc'] = join(' ', $value);
02383                 return $result;
02384             }
02385         }
02386         $done = false;
02387         do {
02388             // this loop checks for type|type|type
02389             // and for type|object classname|type|object classname2
02390             if (strpos($value[0], '|')) {
02391                 $temptypes = explode('|', $value[0]);
02392                 while (count($temptypes)) {
02393                     $type   = array_shift($temptypes);
02394                     $types .= $type;
02395                     if (count($temptypes)) {
02396                         $types .= '|';
02397                     }
02398                 }
02399                 if (trim($type) == 'object') {
02400                     $types .= ' ';
02401                     $this->_removeWhiteSpace($value, 0);
02402                 } else {
02403                     $done = true;
02404                 }
02405                 array_shift($value);
02406                 if (isset ($value[0]) && strlen($value[0]) && ($value[0]{0} == '$' ||
02407                     substr($value[0], 0, 2) == '&$')
02408                 ) {
02409                     // was just passed "object" and the next thing is a variable name
02410                     $result['var']  = trim($value[0]);
02411                     $result['type'] = $types;
02412                     array_shift($value);
02413                     $result['desc'] = join(' ', $value);
02414                     return $result;
02415                 }
02416             } else {
02417                 $types .= $value[0];
02418                 array_shift($value);
02419                 $done = true;
02420             }
02421         } while (!$done && count($value));
02422         $result['type'] = rtrim($types);
02423         $this->_removeWhiteSpace($value, 0);
02424         if ($checkforvar) {
02425             if (!count($value)) {
02426                 $result['var'] = false;
02427             } else {
02428                 /*
02429                  * check for:
02430                  *    variable name ($) 
02431                  *    var passed by reference (&$)
02432                  *    method name (only used by magic method)
02433                  */
02434                 if (substr($value[0], 0, 1) == '$' ||
02435                     substr($value[0], 0, 2) == '&$' ||
02436                     substr($value[0], -2, 2) == '()'
02437                 ) {
02438                     $result['var'] = trim($value[0]);
02439                     array_shift($value);
02440                 } else {
02441                     $result['var'] = false;
02442                 }
02443             }
02444         }
02445         $result['desc'] = join(' ', $value);
02446         return $result;
02447     }
