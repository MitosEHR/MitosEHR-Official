00823     function handleList($word, $pevent)
00824     {
00825         if (is_string($word) && $this->checkEventPush($word, $pevent))
00826         {
00827             return;
00828         }
00829         $ordered = false;
00830         if (!is_object($this->p_vars['last_word']) && strtolower($this->p_vars['last_word']) == '<ol>')
00831         {
00832             // ordered list
00833             $ordered = true;
00834         }
00835         // start a new list
00836         if (!is_object($this->p_vars['last_word']) && (strtolower($this->p_vars['last_word']) == '<ol>' || strtolower($this->p_vars['last_word']) == '<ul>'))
00837         {
00838             $this->p_flags['in_item'] = false;
00839             $this->setContext('list');
00840             $this->p_vars['lists'][++$this->p_vars['list_count']] = new parserList($ordered);
00841         }
00842         if (!is_object($word) && strtolower($word) == '<li>')
00843         {
00844             if ($this->p_flags['in_item'])
00845             {
00846                 // end of a list item (no end tag), add it to the list
00847                 $this->p_vars['lists'][$this->p_vars['list_count']]->addItem($this->p_vars['list_item'][$this->p_vars['list_count']]);
00848                 unset($this->p_vars['list_item'][$this->p_vars['list_count']]);
00849             }
00850             // start a new list item
00851             $this->p_vars['list_item'][$this->p_vars['list_count']] = new parserStringWithInlineTags;
00852             $this->p_flags['in_item'] = true;
00853         } else
00854         {
00855             if (is_object($word) || (strtolower($word) != '</li>'))
00856             {
00857                 if (is_object($word) || (strtolower($word) != '</ul>' && strtolower($word) != '</ol>'))
00858                 {
00859                     // item text
00860                     if (isset($this->p_vars['list_item'][$this->p_vars['list_count']]))
00861                     {
00862                         if (is_string($word) && $word == ' ' &&
00863                               $this->p_vars['last_word'] == ' ') return;
00864                         $this->p_vars['list_item'][$this->p_vars['list_count']]->add($word);
00865                     }
00866                 } else
00867                 {
00868                     if ($this->p_flags['in_item'])
00869                     {
00870                         // end the current list item before ending a list
00871                         $this->p_vars['lists'][$this->p_vars['list_count']]->addItem($this->p_vars['list_item'][$this->p_vars['list_count']]);
00872                         unset($this->p_vars['list_item'][$this->p_vars['list_count']]);
00873                         $this->p_flags['in_item'] = false;
00874                     }
00875                     if (is_string($word) && $this->checkEventPop($word, $pevent))
00876                     {
00877                         if ($this->p_vars['list_count'] > 1)
00878                         {
00879                             // this is a sublist, add it to the list item of the parent list
00880                             if (!isset($this->p_vars['list_item'][$this->p_vars['list_count'] - 1])) {
00881                                 addErrorDie(PDERROR_UL_IN_UL);
00882                             }
00883                             $this->p_vars['list_item'][$this->p_vars['list_count'] - 1]->add($this->p_vars['lists'][$this->p_vars['list_count']]);
00884                             // remove the sublist item and sublist, drop to parent list
00885                             unset($this->p_vars['lists'][$this->p_vars['list_count']]);
00886                             unset($this->p_vars['lists'][$this->p_vars['list_count']]);
00887                             $this->p_vars['list_count']--;
00888                             $this->p_flags['in_item'] = true;
00889                         } else
00890                         {
00891                             // this is a primary list and it has concluded
00892                             $this->pars[$this->p_vars['curpar']]->add($this->p_vars['lists'][$this->p_vars['list_count']]);
00893                             unset($this->p_vars['lists']);
00894                             unset($this->p_vars['list_item']);
00895                             $this->p_vars['list_count'] = 0;
00896                             $this->dropContext();
00897                         }
00898                     }
00899                 }
00900             } else
00901             {
00902                 // check to make sure our list item is not unclosed
00903                 if (!$this->p_flags['in_item'])
00904                 {
00905                     addError(PDERROR_TEXT_OUTSIDE_LI);
00906                 } else
00907                 {
00908                     // end of a list item, add it to the list
00909                     $this->p_vars['lists'][$this->p_vars['list_count']]->addItem($this->p_vars['list_item'][$this->p_vars['list_count']]);
00910                     unset($this->p_vars['list_item'][$this->p_vars['list_count']]);
00911                     $this->p_flags['in_item'] = false;
00912                 }
00913             }
00914         }

01188     function doSimpleList($word)
01189     {
01190         if ($this->p_flags['in_event']) return true;
01191         if (is_object($word)) return false;

01193         if ((strlen($ltrimword) != strlen($word))
01194              && strlen($ltrimword) > 1
01195              && ((in_array($ltrimword{0},array('o','-','1','0','#','+')) && $ltrimword{1} == ' '))
01196                  || ((strlen($ltrimword) >= 2) && (substr($ltrimword,0,2) === '1.' || substr($ltrimword,0,2) === '0.') && $ltrimword{2} == ' '))
01197         {
01198             // save the whitespace for comparison
01199             $this->p_vars['whitespace'] = substr($word,0,strlen($word) - strlen($ltrimword));
01200             $this->p_vars['start_list'] = $ltrimword{0};
01201             if ($this->p_vars['start_list'] != '1' && $this->p_vars['start_list'] != '1.' &&
01202                 $this->p_vars['start_list'] != '0' && $this->p_vars['start_list'] != '0.')
01203             {
01204                 $this->p_flags['orderedlist'] = false;
01205             } else
01206             {
01207                 if (substr($ltrimword,0,2) == '1.')
01208                 {
01209                     $this->p_vars['start_list'] = '1.';
01210                 }
01211                 $this->p_flags['orderedlist'] = true;
01212             }
01213             $this->p_vars['event_stack']->pushEvent(PHPDOCUMENTOR_PDP_EVENT_SIMLIST);
01214             $this->setContext('list');
01215             $this->p_flags['simplelist'] = true;
01216             $this->p_vars['lists'][0] = new parserList($this->p_flags['orderedlist']);
01217             $this->p_vars['list_count'] = 0;
01218             $this->handleSimpleList($word, PHPDOCUMENTOR_PDP_EVENT_SIMLIST, true);
01219             return true;
01220         }
01221         return false;
