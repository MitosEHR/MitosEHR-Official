01407     function _parse_attrs($tag_args)
01408     {
01409 
01410         /* Tokenize tag attributes. */
01411         preg_match_all('/(?:' . $this->_obj_call_regexp . '|' . $this->_qstr_regexp . ' | (?>[^"\'=\s]+)

01419             0 - expecting attribute name

01465                     } else
