01318     function getRootTree($tree, $package, $noparent = false)
01319     {
01320         if (!$tree) { return ''; }
01321         $my_tree = '';
01322         $cur = '#root';
01323         $lastcur = array(false);
01324         $kids = array();
01325         $dopar = false;
01326         if (!$noparent && isset($tree[$cur]['parent']) && $tree[$cur]['parent']) {
01327             $dopar = true;
01328             if (!is_object($tree[$cur]['parent'])) {
01329                 $my_tree .= '<li>' . $tree[$cur]['parent'] .'<ul>';
01330             } else {
01331                 $root = $this->classes->getClassByPackage($tree[$cur]['parent']->name, $package);
01332                 $my_tree .= '<li>' . $this->returnSee($tree[$cur]['parent']);
01333                 if ($tree[$cur]['parent']->package != $package) { $my_tree .= ' <b>(Different package)</b><ul>'; }
01334             }
01335         }
01336         do {
01337             // if class has no children, or has children that are not yet handled
01338             if (!isset($kids[$cur])) {
01339 
01340                 // show the class name itself
01341                 $my_tree .= '<li>'.$this->returnSee($tree[$cur]['link']);
01342 
01343                 // handle interfaces, if there are any
01344                 $root = $this->classes->getClassByPackage(
01345                     (isset($tree[$cur]['link']->name)    ? $tree[$cur]['link']->name    : ''),
01346                     (isset($tree[$cur]['link']->package) ? $tree[$cur]['link']->package : '')
01347                 );
01348                 $interfaces = $root->getImplements();
01349                 if (count($interfaces) > 0) {
01350                     $my_tree .= ' (implements ';
01351                     $needsComma = false;
01352                     foreach ($interfaces as $interface) {
01353                         $my_tree .= $needsComma ? ', ' : '';
01354                         $link = $this->getLink('object ' . $interface, $package);
01355                         if (is_a($link,'abstractLink')) {
01356                             $my_tree .= $this->returnSee($link);
01357                         } else {
01358                             $my_tree .= $interface;
01359                         }
01360                         $needsComma = true;
01361                     }
01362                     unset($needsComma);
01363                     $my_tree .= ')';
01364                 }
01365             }
01366 
01367             if (count($tree[$cur]['children'])) {
01368                 if (!isset($kids[$cur])) {
01369                     $kids[$cur] = 1;
01370                     $my_tree .= '<ul>'."\n";
01371                 }
01372                 array_push($lastcur,$cur);
01373                 list(,$cur) = each($tree[$cur]['children']);
01374                 if ($cur) {
01375                     $cur = $cur['package'] . '#' . $cur['class'];
01376                     continue;
01377                 } else {
01378                     $cur = array_pop($lastcur);
01379                     $cur = array_pop($lastcur); // will fall into infinite loop if this second array_pop() is removed
01380                     $my_tree .= '</ul></li>'."\n";
01381                 }
01382             } else {
01383                 $my_tree .= '</li>';
01384                 $cur = array_pop($lastcur);
01385             }
01386         } while ($cur);
01387         if ($dopar) {
01388             $my_tree .= '</ul></li>'."\n";
01389         }
01390         return $my_tree;
